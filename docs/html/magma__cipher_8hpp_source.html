93 uint32_t f(uint32_t halfBlock, uint32_t roundKey) const {
94 uint32_t temp = halfBlock + roundKey;
95 uint8_t output[4];
96 // Every output element contains two S-blocks
97 output[0] =
98 (sBox[0][(temp >> 28) & 0xF] << 4) | sBox[1][(temp >> 24) & 0xF];
99 output[1] =
100 (sBox[2][(temp >> 20) & 0xF] << 4) | sBox[3][(temp >> 16) & 0xF];
101 output[2] = (sBox[4][(temp >> 12) & 0xF] << 4) | sBox[5][(temp >> 8) & 0xF];
